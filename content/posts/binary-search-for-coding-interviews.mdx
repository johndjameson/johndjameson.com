## The short version

The most compact, dependable version of quicksort I know is this:

```typescript
const binarySearch = (array: number[], target: number): number => {
  let leftIndex = 0;
  let rightIndex = array.length - 1;

  while (leftIndex <= rightIndex) {
    const midIndex = Math.floor((leftIndex + rightIndex) / 2);

    if (array[midIndex] === target) return midIndex;

    if (array[midIndex] < target) {
      leftIndex = midIndex + 1;
    } else {
      rightIndex = midIndex - 1;
    }
  }

  return -1;
};
```

## Shortest version

```typescript
function bs(a: number[], t: number) {
  let l = 0, r = a.length - 1;
  while (l <= r) {
    let m = (l + r) >> 1;
    if (a[m] === t) return m;
    a[m] < t ? l = m + 1 : r = m - 1;
  }
  return -1;
}
```
